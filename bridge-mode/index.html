еҰӮжһңиҰҒж–°еўһдёҖдёӘе“ҒзүҢпјҢжҲ–иҖ…еҠ дёҖдёӘж–°зҡ„еј•ж“ҺпјҲжҜ”еҰӮж ёеҠЁеҠӣпјүпјҢйӮЈд№Ҳеӯҗзұ»зҡ„ж•°йҮҸеўһй•ҝжӣҙеҝ«гҖӮ

жүҖд»ҘпјҢжЎҘжҺҘжЁЎејҸе°ұжҳҜдёәдәҶйҒҝе…ҚзӣҙжҺҘз»§жүҝеёҰжқҘзҡ„еӯҗзұ»зҲҶзӮёгҖӮ

жҲ‘д»¬жқҘзңӢзңӢжЎҘжҺҘжЁЎејҸеҰӮдҪ•и§ЈеҶідёҠиҝ°й—®йўҳгҖӮ

еңЁжЎҘжҺҘжЁЎејҸдёӯпјҢйҰ–е…ҲжҠҠCarжҢүе“ҒзүҢиҝӣиЎҢеӯҗзұ»еҢ–пјҢдҪҶжҳҜпјҢжҜҸдёӘе“ҒзүҢйҖүжӢ©д»Җд№ҲеҸ‘еҠЁжңәпјҢдёҚеҶҚдҪҝз”Ёеӯҗзұ»жү©е……пјҢиҖҢжҳҜйҖҡиҝҮдёҖдёӘжҠҪиұЎзҡ„вҖңдҝ®жӯЈвҖқзұ»пјҢд»Ҙз»„еҗҲзҡ„еҪўејҸеј•е…ҘгҖӮжҲ‘д»¬жқҘзңӢзңӢе…·дҪ“зҡ„е®һзҺ°гҖӮ

йҰ–е…Ҳе®ҡд№үжҠҪиұЎзұ»CarпјҢе®ғеј•з”ЁдёҖдёӘEngineпјҡ

public abstract class Car {
    // еј•з”ЁEngine:
    protected Engine engine;

    public Car(Engine engine) {
        this.engine = engine;
    }

    public abstract void drive();
}

Engineзҡ„е®ҡд№үеҰӮдёӢпјҡ

public interface Engine {
    void start();
}

зҙ§жҺҘзқҖпјҢеңЁдёҖдёӘвҖңдҝ®жӯЈвҖқзҡ„жҠҪиұЎзұ»RefinedCarдёӯе®ҡд№үдёҖдәӣйўқеӨ–ж“ҚдҪңпјҡ

public abstract class RefinedCar extends Car {
    public RefinedCar(Engine engine) {
        super(engine);
    }

    public void drive() {
        this.engine.start();
        System.out.println("Drive " + getBrand() + " car...");
    }

    public abstract String getBrand();
}

иҝҷж ·дёҖжқҘпјҢжңҖз»Ҳзҡ„дёҚеҗҢе“ҒзүҢз»§жүҝиҮӘRefinedCarпјҢдҫӢеҰӮBossCarпјҡ

public class BossCar extends RefinedCar {
    public BossCar(Engine engine) {
        super(engine);
    }

    public String getBrand() {
        return "Boss";
    }
}

иҖҢй’ҲеҜ№жҜҸдёҖз§Қеј•ж“ҺпјҢз»§жүҝиҮӘEngineпјҢдҫӢеҰӮHybridEngineпјҡ

public class HybridEngine implements Engine {
    public void start() {
        System.out.println("Start Hybrid Engine...");
    }
}

е®ўжҲ·з«ҜйҖҡиҝҮиҮӘе·ұйҖүжӢ©дёҖдёӘе“ҒзүҢпјҢеҶҚй…ҚеҗҲдёҖз§Қеј•ж“ҺпјҢеҫ—еҲ°жңҖз»Ҳзҡ„Carпјҡ

RefinedCar car = new BossCar(new HybridEngine());
car.drive();

дҪҝз”ЁжЎҘжҺҘжЁЎејҸзҡ„еҘҪеӨ„еңЁдәҺпјҢеҰӮжһңиҰҒеўһеҠ дёҖз§Қеј•ж“ҺпјҢеҸӘйңҖиҰҒй’ҲеҜ№Engineжҙҫз”ҹдёҖдёӘж–°зҡ„еӯҗзұ»пјҢеҰӮжһңиҰҒеўһеҠ дёҖдёӘе“ҒзүҢпјҢеҸӘйңҖиҰҒй’ҲеҜ№RefinedCarжҙҫз”ҹдёҖдёӘеӯҗзұ»пјҢд»»дҪ•RefinedCarзҡ„еӯҗзұ»йғҪеҸҜд»Ҙе’Ңд»»дҪ•дёҖз§ҚEngineиҮӘз”ұз»„еҗҲпјҢеҚідёҖиҫҶжұҪиҪҰзҡ„дёӨдёӘз»ҙеәҰпјҡе“ҒзүҢе’Ңеј•ж“ҺйғҪеҸҜд»ҘзӢ¬з«Ӣең°еҸҳеҢ–гҖӮ
